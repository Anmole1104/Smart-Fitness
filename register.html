function register(event) { event.preventDefault(); const username = document.getElementById("username").value; const password = document.getElementById("password").value; localStorage.setItem(username, password); alert("Registered successfully!"); window.location.href = "login.html"; } function login(event) { event.preventDefault(); const username = document.getElementById("username").value; const password = document.getElementById("password").value; const storedPassword = localStorage.getItem(username); if (storedPassword === password) { localStorage.setItem("loggedIn", "true"); alert("Login successful!"); window.location.href = "index.html"; } else { alert("Incorrect credentials!"); } }
